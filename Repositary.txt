public class Repositary{

	public String[] saves=new String[2];
	private int position;
	
	public void save(String name)
	{
		if(position<2)
		{
			System.out.println(name +" is valid position no"+ position);
			this.saves[position]=name;
			position++;
		}
		else
		{
			system.out.println("out of storage");
		}
	}
	
	public static void main(String[] args)
	{
		Repositary repos=new Repositary();
		repos.save("rohan a");
		repos.save("rohan b");
	}

}
